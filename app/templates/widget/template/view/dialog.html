                        <div class="mx-mask"></div>

                            </div>
                        </div>
                    </div>
            </div>
            <script>
                 $(function(){
                    var $iosDialog1 = $('#iosDialog1');
                    $('#dialogs').on('click', '.mx-dialog__btn', function(){
                        $(this).parents('.js_dialog').fadeOut(200);
                    });
                    $('#showIOSDialog1').on('click', function(){
                        $iosDialog1.fadeIn(200);
                    });
                 });
             </script>
